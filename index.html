2. Meaningful Names

  • Meaningful, intention-revealing, pronounceable
  • Avoid disinformation
  • Nouns for classes, verbs for methods
  • Do not hesitate to rename poor names globally

self.getThem = function(theList) {
    var list1 = [];
    for (var i = 0; i < theList.length; i++) {
        if (theList[i] === 4) {
            list1.push(theList[i]);
        }
    }
    return list1;
};
self.getFlaggedCells = function(gameBoard) {
    var flaggedCells = [];
    $.each(gameBoard, function(index, currentCell) {
        if (currentCell.status === 'FLAGGED') {
            flaggedCells.push(currentCell);
        }
    });
    return flaggedCells;
};
